CMAKE_MINIMUM_REQUIRED(VERSION 2.6)
PROJECT(dali-demo C CXX)

SET(dali-demo_VERSION_MAJOR 1)
SET(dali-demo_VERSION_MINOR 0)

SET(ROOT_SRC_DIR ${CMAKE_SOURCE_DIR}/../..)

SET(PREFIX ${CMAKE_INSTALL_PREFIX})
SET(BINDIR "${PREFIX}/bin")
SET(APP_DATA_PREFIX ${PREFIX}/share/com.samsung.dali-demo)

SET(LOCAL_IMAGES_DIR ${ROOT_SRC_DIR}/demo/images)
SET(LOCAL_MODELS_DIR ${ROOT_SRC_DIR}/demo/models)
SET(LOCAL_SCRIPTS_DIR ${ROOT_SRC_DIR}/demo/scripts)

SET(IMAGES_DIR ${APP_DATA_PREFIX}/images/)
SET(MODELS_DIR ${APP_DATA_PREFIX}/models/)
SET(SCRIPTS_DIR ${APP_DATA_PREFIX}/scripts/)

SET(DALI_IMAGE_DIR \\"${IMAGES_DIR}\\")
SET(DALI_MODEL_DIR \\"${MODELS_DIR}\\")
SET(DALI_SCRIPT_DIR \\"${SCRIPTS_DIR}\\")
SET(DALI_EXAMPLE_BIN \\"${BINDIR}\\")

FILE(GLOB LOCAL_IMAGES_LIST RELATIVE "${LOCAL_IMAGES_DIR}" "${LOCAL_IMAGES_DIR}/*")
FOREACH(flag ${LOCAL_IMAGES_LIST})
        SET(LOCAL_IMAGES "${LOCAL_IMAGES} ${flag}")
ENDFOREACH(flag)
INSTALL(FILES ${LOCAL_IMAGES} DESTINATION ${IMAGES_DIR})

#FILE(GLOB LOCAL_MODELS RELATIVE ${LOCAL_MODELS_DIR} *)
#INSTALL(FILES ${LOCAL_MODELS} DESTINATION ${MODELS_DIR})

#FILE(GLOB LOCAL_SCRIPTS RELATIVE ${LOCAL_MODELS_DIR} *)
#INSTALL(FILES ${LOCAL_SCRIPTS} DESTINATION ${SCRIPTS_DIR})

SET(PKG_LIST dali
             dali-toolkit)

INCLUDE(FindPkgConfig)
pkg_check_modules(REQUIRED_PKGS REQUIRED ${PKG_LIST})

FOREACH(flag ${REQUIRED_PKGS_CFLAGS})
        SET(REQUIRED_CFLAGS "${REQUIRED_CFLAGS} ${flag}")
ENDFOREACH(flag)

SET(DALI_DEMO_CFLAGS "-DDALI_IMAGE_DIR=${DALI_IMAGE_DIR} -DDALI_MODEL_DIR=${DALI_MODEL_DIR} -DDALI_SCRIPT_DIR=${DALI_SCRIPT_DIR} -DDALI_EXAMPLE_BIN=${DALI_EXAMPLE_BIN}")

SET(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${REQUIRED_CFLAGS} ${DALI_DEMO_CFLAGS} -Werror -Wall")
SET(CMAKE_CXX_FLAGS "${CMAKE_C_FLAGS}")

INCLUDE_DIRECTORIES(${ROOT_SRC_DIR})
INCLUDE_DIRECTORIES(${DEMO_SRC_DIR})

ADD_SUBDIRECTORY(demo)
ADD_SUBDIRECTORY(examples)
